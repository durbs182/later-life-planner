#!/usr/bin/env bash
set -euo pipefail


LOGDIR="./deletion-log-$(date +%Y%m%d%H%M%S)"
mkdir -p "$LOGDIR"

while IFS= read -r line; do
  displayName=$(echo "$line" | awk '{print $1}')
  appId=$(echo "$line" | awk '{print $2}')
  appObjectId=$(echo "$line" | awk '{print $3}')

  echo "Deleting app: $displayName  appId:$appId  objectId:$appObjectId"
  echo "Backing up manifest..."
  az rest --method GET --uri "https://graph.microsoft.com/v1.0/applications/${appObjectId}" -o json > "${LOGDIR}/${displayName}_${appObjectId}_manifest.json"

  # Delete service principal(s) for this app (if any)
  sp_ids=$(az rest --method GET --uri "https://graph.microsoft.com/v1.0/servicePrincipals?\$filter=appId eq '${appId}'&\$select=id,displayName" -o json | jq -r '.value[]?.id')
  for sp in $sp_ids; do
    echo "Deleting servicePrincipal id=$sp"
    az rest --method DELETE --uri "https://graph.microsoft.com/v1.0/servicePrincipals/${sp}" || echo "Failed to delete servicePrincipal ${sp}, continuing"
  done

  # Delete the application object
  az rest --method DELETE --uri "https://graph.microsoft.com/v1.0/applications/${appObjectId}" || echo "Failed to delete application ${appObjectId}, continuing"

  echo "Deleted $displayName" >> "${LOGDIR}/deleted_apps.log"
done < apps-to-delete.txt

echo "Done. Logs and manifests in $LOGDIR"
